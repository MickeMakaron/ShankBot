52 void AppearancesReader::readAppearances(std::string path)
53 {
54  std::ifstream file(path, std::ios::binary);
55  if(!file.good())
56  {
57  return;
58  }
59 
60  file.seekg(0, file.end);
61  size_t fileSize = file.tellg();
62  file.seekg(0);
63 
64  if(fileSize == 0)
65  {
66  return;
67  }
68 
69  Buffer buffer;
70  buffer.data.resize(fileSize);
71  file.read(&buffer.data[0], fileSize);
72  file.close();
73 
74  while(buffer.curr < buffer.data.size())
75  {
76  mObjects.push_back(readObject(buffer));
77  }
78 }
