52 void CatalogContent::read(std::string filePath)
53 {
54  using namespace sb::utility::file;
55  if(!fileExists(filePath))
56  SB_THROW("Cannot open catalog content at '" + filePath + "'. File does not exist.");
57 
58 
59  QFile file(QString::fromStdString(filePath));
60  file.open(QIODevice::ReadOnly | QIODevice::Text);
61  QString content = file.readAll();
62  file.close();
63  QJsonDocument catalogContent = QJsonDocument::fromJson(content.toUtf8());
64 
65  if(catalogContent.isNull())
66  SB_THROW("Failed to read catalog content at '" + filePath + "'.");
67 
68  QJsonArray array = catalogContent.array();
69  std::string dir = getPath(filePath);
70  for(const QJsonValue& val : array)
71  {
72  QJsonObject obj = val.toObject();
73  QString type = obj["type"].toString();
74 
75  if(type == "appearances")
76  {
77  Appearances a;
78  a.path = dir + "/" + obj["file"].toString().toStdString();
79  mAppearances.push_back(a);
80  }
81  else if(type == "sprite")
82  {
83  SpriteSheet s;
84  s.path = dir + "/" + obj["file"].toString().toStdString();
85  s.spriteSize = (SpriteSize)obj["spritetype"].toInt();
86  s.firstSpriteId = obj["firstspriteid"].toInt();
87  s.lastSpriteId = obj["lastspriteid"].toInt();
88  s.area = obj["area"].toInt();
89  mSpriteSheets.push_back(s);
90  }
91  else
92  {
93  SB_THROW("Error reading catalog content. Unimplemented type.");
94  }
95  }
96 }
