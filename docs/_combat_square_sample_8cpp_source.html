67 std::vector<CombatSquareSample::CombatSquare> CombatSquareSample::generateSamples()
68 {
69  int argc = 0;

71  QImage img(CombatSquare::width, CombatSquare::height, QImage::Format_RGBA8888);
72  img.fill(Qt::transparent);
73  QPainter painter(&img);
74  QPainterPath outPath;
75  outPath.addRect(0, 0, img.width() - 1, img.height() - 1);
76  QPainterPath inPath;
77  inPath.addRect(2, 2, img.width() - 5, img.height() - 5);
78 
79  std::vector<CombatSquare> combatSquares;
80  combatSquares.reserve(M_GENERATION_DETAILS.size());
81  for(const GenerationDetails& d : M_GENERATION_DETAILS)
82  {
83  painter.strokePath(inPath, QPen(QColor(d.in.r, d.in.g, d.in.b, d.in.a), 3));
84  painter.strokePath(outPath, QPen(QColor(d.out.r, d.out.g, d.out.b, d.out.a), 3));
85  combatSquares.emplace_back(d.type, img.constBits());
86  }
87 
88  return combatSquares;
