55 ConnectionManager::ConnectionId ConnectionManager::poll(Event& event, size_t timeOut)
56 {
57  DWORD result = WaitForMultipleObjects(mEvents.size(), mEvents.data(), FALSE, timeOut);
58 
59  if(result >= WAIT_OBJECT_0 && result < WAIT_OBJECT_0 + mEvents.size())
60  {
61  size_t i = result - WAIT_OBJECT_0;
62  assert(i < mEvents.size());
63  assert(i < mConnections.size());
64 
65  Connection& connection = *mConnections[i];
66  typedef Connection::State S;
67  S previousState = connection.getState();
68  connection.update();
69 
70  S currentState = connection.getState();
71 
72  if(currentState == S::DROPPED)
73  {
74  event = Event::DROP;
75  return i;
76  }
77 
78  switch(previousState)
79  {
80  case S::READING: event = Event::READ; break;
81  case S::WRITING: event = Event::WRITE; break;
82  case S::CONNECTING: event = Event::CONNECT; break;
83  case S::DROPPED: event = Event::DROP; break;
84 
85  default:
86  SB_THROW("Unexpected state transition in message connection. From ", (int)previousState, " to ", (int)currentState);
87  }
88  return i;
89  }
90 
91  if(result == WAIT_TIMEOUT)
92  {
93  event = Event::TIME_OUT;
94  return -1;
95  }
96 
97 
98  DWORD error = GetLastError();
99  switch(error)
100  {
101  case ERROR_INVALID_PARAMETER: // This happens if mEvents.size() == 0
102  event = Event::INVALID_STATE;
103  return -1;
104 
105  default:
106  SB_THROW("Unexpected result when waiting for message connection events: ", result, ". Error: ", error);
107  }
108 }
