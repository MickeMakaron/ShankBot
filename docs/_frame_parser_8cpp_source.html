825  std::shared_ptr<Matrix<float, 4, 4>> transform = std::make_shared<Matrix<float, 4, 4>>(program.transform);

836  float bot = std::numeric_limits<float>::min();

839  assert(size_t(&vertices[indices[i]]) < bufferEnd);

874  r.color = color;
875  r.hasOrder = true;
876  assert(size_t(&orders[indices[i]]) < bufferEnd);
877  r.order = orders[indices[i]];
878  r.drawCallId = mDrawCallId;
879  r.isDepthTestEnabled = drawCall.isDepthTestEnabled;
880  r.isDepthWriteEnabled = drawCall.isDepthWriteEnabled;
881 
882  i += 2;
883  }
