50 void GraphicsMonitorReader::waitForFrame() const
51 {
52  switch(WaitForSingleObject(mShm->semRead, 500))
53  {
54  case WAIT_ABANDONED:
55  SB_THROW("Wait abandoned.");
56 
57  case WAIT_OBJECT_0:
58  return;
59 
60  case WAIT_TIMEOUT:
61  if(!mClient.isAlive())
62  SB_THROW("Tibia client unexpectedly terminated.");
63 
64  mClient.getInput().sendPaintMessage();
65  waitForFrame();
66  return;
67 
68  case WAIT_FAILED:
69  {
70  SB_THROW("Failed wait. Error code: ", GetLastError(), "\n");
71  }
72 
73  default:
74  SB_THROW("Unimplemented wait signal.");
75  }
76 }
