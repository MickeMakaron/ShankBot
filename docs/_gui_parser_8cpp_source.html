51 void GuiParser::parse(const Frame& frame)
52 {
53  mData = Data();
54  pass1 = Pass1();
55  mDraws = frame.guiDraws;
56  mBaseNames.clear();
57 
58  if(mDraws == nullptr)
59  {
60  return;
61  }
62 
63  mHalfFrameWidth = frame.width / 2.f;
64  mHalfFrameHeight = frame.height / 2.f;
65 
66  mBaseNames.resize(mDraws->size());
67  std::transform(mDraws->begin(), mDraws->end(), mBaseNames.begin(), [](const GuiDraw& d)
68  {
69  return file::basenameNoExt(d.name);
70  });
71 
72  for(size_t i = 0; i < mDraws->size(); i++)
73  {
74  auto foundIt = mGuiDrawHandlers.find(mBaseNames[i]);
75  if(foundIt != mGuiDrawHandlers.end())
76  {
77  try
78  {
79  foundIt->second(i);
80  }
81  catch(const std::runtime_error& e)
82  {
83  std::cout << "Failed to handle GUI draw: \"" << e.what() << "\"." << std::endl;
84  }
85  }
86  else
87  {
88 // std::cout << "Unhandled: " << mBaseNames[i] << std::endl;
89  }
90  }
91 
92  try
93  {
94  parsePass1();
95  }
96  catch(const std::runtime_error& e)
97  {
98  std::cout << "Failed to parse pass 1: \"" << e.what() << "\"." << std::endl;
99  }
100 }

102 void GuiParser::parsePass1()
103 {
104  if(pass1.hpManaBorders.empty())
105  {
106  return;
107  }
108  SB_EXPECT(pass1.hpManaBorders.size(), ==, 2);
109 
110  const GuiDraw* border = pass1.hpManaBorders[0].draw;
111  float borderWidth = border->botRight.x - border->topLeft.x;
112 
113  const GuiDraw* mana = pass1.manaFill.draw;
114  const GuiDraw* hp = pass1.hpFill.draw;
115 
116  DefaultSideBar::StatusBar& sb = mData.game.defaultSideBar.statusBar;
117  sb.manaPercent = (mana ? mana->botRight.x - mana->topLeft.x : 0.f) / borderWidth;
118  sb.hpPercent = (hp ? hp->botRight.x - hp->topLeft.x : 0.f) / borderWidth;

129 IRect GuiParser::getRect(Vertex topLeft, Vertex botRight)
130 {
131  if(botRight.x < topLeft.x || botRight.y < topLeft.y)
132  {
133  std::swap(topLeft, botRight);
134  }
135 
136  IRect ir;
137  ir.x = round(topLeft.x);
138  ir.y = round(topLeft.y);
139  ir.width = round(botRight.x - topLeft.x);
140  ir.height = round(botRight.y - topLeft.y);
141 
142  return ir;
143 }

145 IRect GuiParser::getScreenRect(const Draw& d)
146 {
147  Vertex topLeft;
148  Vertex botRight;
149  d.getScreenCoords(mHalfFrameWidth, mHalfFrameHeight, topLeft.x, topLeft.y, d.topLeft.x, d.topLeft.y);
150  d.getScreenCoords(mHalfFrameWidth, mHalfFrameHeight, botRight.x, botRight.y, d.botRight.x, d.botRight.y);
