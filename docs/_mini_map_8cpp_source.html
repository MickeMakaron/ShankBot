88 bool MiniMap::parseMiniMapColorFileName(const std::string& fileName, unsigned int& x, unsigned int& y, unsigned char& level)
89 {
90  static const size_t X_INDEX = M_MINI_MAP_COLOR_FILE_PREFIX.size();
91 
92  std::string base = basename(fileName);
93  if(base.find(M_MINI_MAP_COLOR_FILE_PREFIX) != 0)
94  return false;
95 
96  size_t separatorIndex = base.find(M_MINI_MAP_FILE_SEPARATOR, X_INDEX);
97  assert(separatorIndex != base.npos);
98  std::string xStr(&base[X_INDEX], &base[separatorIndex]);
99  assert(isNumeric(xStr));
100 
101  size_t yIndex = separatorIndex + 1;
102  separatorIndex = base.find(M_MINI_MAP_FILE_SEPARATOR, yIndex);
103  assert(separatorIndex != base.npos);
104  std::string yStr(&base[yIndex], &base[separatorIndex]);
105  assert(isNumeric(yStr));
106 
107  size_t levelIndex = separatorIndex + 1;
108  separatorIndex = base.find('.', levelIndex);
109  assert(separatorIndex != base.npos);
110  std::string levelStr(&base[levelIndex], &base[separatorIndex]);
111  assert(isNumeric(levelStr));
112 
113  x = strToInt(xStr);
114  y = strToInt(yStr);
115  level = strToInt(levelStr);
116  return true;
117 }
