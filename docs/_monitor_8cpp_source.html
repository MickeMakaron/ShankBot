466 void Monitor::updateSharedMemory()
467 {
468  if(mDataBuffer.size() < sizeof(Frame))
469  {
470  return;
471  }
472 
473  size_t insertionSize = 0;
474  if(mDataBuffer.size() > DATA_BUFFER_SIZE)
475  {
476  while(true)
477  {
478  const Frame& f = *(Frame*)((char*)mDataBuffer.data() + insertionSize);
479  size_t newInsertionSize = insertionSize + f.size;
480  if(newInsertionSize > DATA_BUFFER_SIZE || newInsertionSize > mDataBuffer.size())
481  break;
482  insertionSize = newInsertionSize;
483  }
484  THROW_ASSERT(insertionSize <= mDataBuffer.size());
485  }
486  else
487  {
488  insertionSize = mDataBuffer.size();
489  }
490 
491  memcpy(mShm->data, mDataBuffer.data(), insertionSize);
492  mShm->size = insertionSize;
493  if(insertionSize == mDataBuffer.size())
494  {
495  mDataBuffer.clear();
496  }
497  else
498  {
499  mDataBuffer.erase(mDataBuffer.begin(), mDataBuffer.begin() + insertionSize);
500  }
501 }
