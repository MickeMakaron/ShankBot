136 std::vector<std::shared_ptr<Sprite>> OutfitAddonMerger::processSprite(unsigned int id, size_t width, size_t height, const unsigned char* pixels)
137 {
138  auto mergersIt = mMerges.find(id);
139  if(mergersIt == mMerges.end())
140  return {};
141 
142  std::shared_ptr<Sprite> sprite = std::make_shared<Sprite>(id, width, height, pixels);
143  std::vector<std::shared_ptr<Sprite>> merges;
144  for(const std::shared_ptr<SnapshotMerger>& m : mergersIt->second)
145  {
146  m->process(sprite);
147  if(m->isReady())
148  {
149  std::vector<std::shared_ptr<Sprite>> newMerges = m->merge();
150  merges.insert(merges.end(), newMerges.begin(), newMerges.end());
151  }
152  }
153 
154  mMerges.erase(mergersIt);
155 
156  return merges;
157 }

165 std::vector<OutfitAddonMerger::SnapshotMerger::OutfitSnapshot> OutfitAddonMerger::getOutfitSnapshots(const Object& outfit) const
166 {
167  assert(outfit.type == Object::Type::OUTFIT);
168 
169  std::vector<SnapshotMerger::OutfitSnapshot> snapshots;
170  for(const Object::SomeInfo& someInfo : outfit.someInfos)
171  {
172  const Object::SpriteInfo& info = someInfo.spriteInfo;
173  size_t colSize = info.numAddons;
174  size_t colStride = info.numBlendFrames;
175  size_t rowSize = info.numBlendFrames * info.numDirections;
176  size_t rowStride = rowSize * (colSize - 1);
177 
178  snapshots.reserve(snapshots.size() + info.spriteIds.size() / (colSize * colStride));
179  for(size_t i = 0; i < info.spriteIds.size(); i += rowStride)
180  {
181  for(size_t x = 0; x < rowSize; x += colStride, i += colStride)
182  {
183  snapshots.emplace_back();
184  std::vector<size_t>& ids = snapshots.back().spriteIds;
185  ids.reserve(colSize);
186  for(size_t y = 0; y < colSize; y++)
187  ids.push_back(info.spriteIds[i + y * rowSize]);
188  }
189  }
190 
191  }
192  return snapshots;
193 }
