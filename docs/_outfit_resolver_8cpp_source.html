223 std::list<OutfitResolver::Collision> OutfitResolver::removeCollisions(std::list<Text>& names, std::list<TextHpPair>& pairs) const
224 {
225  std::list<Collision> collisions;
226  const short VIEW_RIGHT = mCurrentFrame->viewX + mCurrentFrame->viewWidth;
227  std::function<bool(const Text& t)> addConditionLeft = [](const Text& t){return t.x == 0;};
228  std::function<bool(const Text& t)> addConditionRight = [VIEW_RIGHT](const Text& t){return t.x + t.width == VIEW_RIGHT;};
229  for(auto it1 = pairs.begin(); it1 != pairs.end();)
230  {
231  const Text& text = it1->text;
232  if(text.x == 0 || text.x + text.width == VIEW_RIGHT)
233  {
234  Collision collision;
235 
236 
237  auto addCondition = text.x == 0 ? addConditionLeft : addConditionRight;
238  auto it2 = it1;
239  it2++;
240  while(it2 != pairs.end())
241  {
242  if(addCondition(it2->text))
243  {
244  collision.pairs.push_back(*it2);
245  pairs.erase(it2++);
246  }
247  else
248  it2++;
249  }
250  for(auto nameIt = names.begin(); nameIt != names.end();)
251  {
252  if(addCondition(*nameIt))
253  {
254  collision.names.push_back(*nameIt);
255  names.erase(nameIt++);
256  }
257  else
258  nameIt++;
259  }
260 
261  if(!collision.pairs.empty())
262  {
263  collision.pairs.push_back(*it1);
264  pairs.erase(it1++);
265  collisions.push_back(collision);
266  }
267  else
268  it1++;
269  }
270  else
271  it1++;
272  }
273 
274  return collisions;
275 }

277 std::list<OutfitResolver::TextHpPair> OutfitResolver::pairHpBarsWithNames(std::list<Text>& names, const std::list<HpBar>& hpBars) const
278 {
279  std::list<TextHpPair> pairs;
280  const unsigned short VIEW_RIGHT = mCurrentFrame->viewX + mCurrentFrame->viewWidth;
281  for(auto barIt = hpBars.begin(); barIt != hpBars.end(); barIt++)
282  {
283  const HpBar& b = *barIt;
284  unsigned short minDX = -1;
285  auto closestName = names.end();
286  const unsigned short BAR_MIDDLE_X = b.x + b.width / 2;
287  const unsigned short BAR_RIGHT = b.x + b.width;
288  for(auto nameIt = names.begin(); nameIt != names.end(); nameIt++)
289  {
290  short dY = nameIt->y + nameIt->height - b.y;
291  if(dY <= MAX_Y_DISTANCE && dY >= -MAX_Y_DISTANCE)
292  {
293  short dX;
294  if(nameIt->x == 0)
295  dX = b.x;
296  else if(nameIt->x + nameIt->width == VIEW_RIGHT)
297  dX = VIEW_RIGHT - BAR_RIGHT;
298  else
299  dX = nameIt->x + nameIt->width / 2 - BAR_MIDDLE_X;
300 
301  if(dX < 0) dX = -dX;
302  if(dX < minDX)
303  {
304  minDX = dX;
305  closestName = nameIt;
306  }
307 
308  }
309  }
310 
311  assert(closestName != names.end());
312  TextHpPair pair;
313  pair.text = *closestName;
314  pair.hp = b;
315  pairs.push_back(pair);
316  names.erase(closestName);
317  }
318 
319  assert(pairs.size() == hpBars.size());
320  return pairs;
321 }

324 std::list<Scene::Object>::iterator OutfitResolver::findMatchingOutfit(short left, short right, short bottom, std::list<Scene::Object>& visibleOutfits) const
325 {
326  auto isMatchY = [bottom](const Scene::Object& o)
327  {
328  short dY = o.screenY - bottom;
329  return dY <= MAX_Y_DISTANCE && dY >= -MAX_Y_DISTANCE;
330  };
331 
332 
333  const short VIEW_LEFT = mCurrentFrame->viewX;
334  const short VIEW_RIGHT = VIEW_LEFT + mCurrentFrame->viewWidth;
335  const float TILE_WIDTH = mCurrentTileWidth;
336  auto matchingOutfit = visibleOutfits.end();
337  if(left == 0)
338  {
339  short minX = VIEW_RIGHT;
340  const short LOWER_BOUND_X = VIEW_LEFT - TILE_WIDTH;
341  for(auto outfitIt = visibleOutfits.begin(); outfitIt != visibleOutfits.end(); outfitIt++)
342  {
343  if(isMatchY(*outfitIt))
344  {
345  if(outfitIt->screenX < minX && outfitIt->screenX > LOWER_BOUND_X)
346  {
347  minX = outfitIt->screenX;
348  matchingOutfit = outfitIt;
349  }
350  }
351  }
352 
353  }
354  else if(right == VIEW_RIGHT)
355  {
356  short maxX = VIEW_LEFT;
357  const short UPPER_BOUND_X = VIEW_RIGHT;
358  for(auto outfitIt = visibleOutfits.begin(); outfitIt != visibleOutfits.end(); outfitIt++)
359  {
360  if(isMatchY(*outfitIt))
361  {
362  if(outfitIt->screenX > maxX && outfitIt->screenX < UPPER_BOUND_X)
363  {
364  maxX = outfitIt->screenX;
365  matchingOutfit = outfitIt;
366  }
367  }
368  }
369  }
370  else
371  {
372  short outfitX = left + ((right - left) - TILE_WIDTH) / 2;
373  for(auto outfitIt = visibleOutfits.begin(); outfitIt != visibleOutfits.end(); outfitIt++)
374  {
375  short dX = outfitIt->screenX - outfitX;
376  if(dX < 0) dX = -dX;
377  if(isMatchY(*outfitIt) && dX <= MAX_X_DISTANCE)
378  {
379  matchingOutfit = outfitIt;
380  break;
381  }
382  }
383  }
384 
385  return matchingOutfit;
386 }

388 void OutfitResolver::processNpcs(const std::list<Text>& names, std::list<Scene::Object>& visibleOutfits)
389 {
390  const float TILE_WIDTH = mCurrentTileWidth;
391  for(const Text& npcName : names)
392  {
393  auto matchingOutfit = findMatchingOutfit(npcName.x, npcName.x + npcName.width, npcName.y + npcName.height, visibleOutfits);
394  auto result = mOutfits.insert(std::make_pair(npcName.string, Outfit(npcName.string, Outfit::Type::NPC)));
395  Outfit& storedOutfit = result.first->second;
396  if(!result.second)
397  assert(storedOutfit.type == Outfit::Type::NPC);
398 
399  Npc npc;
400  npc.name = npcName.string;
401  if(matchingOutfit != visibleOutfits.end())
402  {
403  if(storedOutfit.object != -1)
404  assert(storedOutfit.object == matchingOutfit->object);
405  else
406  storedOutfit.object = matchingOutfit->object;
407 
408  npc.x = matchingOutfit->screenX;
409  npc.y = matchingOutfit->screenY;
410  visibleOutfits.erase(matchingOutfit);
411  }
412  else
413  {
414  npc.x = npcName.x + (npcName.width - TILE_WIDTH) / 2;
415  npc.y = npcName.y + npcName.height;
416  }
417

419  mNpcs.push_back(npc);

557 
558  unsigned short cmX = pc.x + mCurrentTileWidth / 2;
559  if(cmX <= VIEW_LEFT)
560  cmX = VIEW_LEFT + 2;
561  else if(cmX >= VIEW_RIGHT)
562  cmX = VIEW_RIGHT - 2;
563 
564  unsigned short cmY = pc.y + mCurrentTileHeight / 2;
565  if(cmY <= VIEW_TOP)
566  cmY = VIEW_TOP + 2;
567  else if(cmY >= VIEW_BOTTOM)
568  cmY = VIEW_BOTTOM -2;
569  std::vector<std::string> contextMenu = getContextMenu(reader, cmX, cmY);
570 
571  bool isOutfit = false;
572  bool isPlayer = false;
573  for(const std::string& option : contextMenu)
574  {
575  if(option == "Copy Name")
576  isOutfit = true;
577  else if(option.find("Message to") != option.npos)
578  {
579  isPlayer = true;
580  break;
581  }
582  }
583 
584  if(isPlayer)
585  {
586  Player p;
587  p.x = pc.x;
588  p.y = pc.y;
589  p.hp = pc.hp;
590  p.name = pc.name;
591  auto result = mOutfits.emplace(p.name, Outfit(p.name, Outfit::Type::PLAYER));
592  assert(result.second);
593 
594  mPlayers.push_back(p);
595  }
596  else if(isOutfit)
597  {
598  Creature c;
599  c.x = pc.x;
600  c.y = pc.y;
601  c.hp = pc.hp;
602  c.name = pc.name;
603  c.object = pc.object;

606 
607  mCreatures.push_back(c);
608  }
609  else
610  {
611  // We probably missed the outfit when getting the context menu.
612  // It may happen during lag spikes.
613  mUnknowns.push_back(pc);
614  }
615  }
