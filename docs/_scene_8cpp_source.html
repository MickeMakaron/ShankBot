93 void Scene::parseCurrentFrame()
94 {
95 // if(mIsCurrentFrameParsed)
96 // return;
97 
98  for(size_t x = 0; x < MAX_VISIBLE_TILES_X; x++)
99  for(size_t y = 0; y < MAX_VISIBLE_TILES_Y; y++)
100  {
101  Tile& tile = mData->tiles[x][y];
102  tile = Tile();
103  tile.tileX = x - VISIBILITY_OFFSET_LOW;
104  tile.tileY = y - VISIBILITY_OFFSET_LOW;
105  }
106 
107  const float VIEW_TEX_TO_SCREEN_RATIO_X = mSceneWidth / ((float)Constants::VIEW_PIXEL_WIDTH);
108  const float VIEW_TEX_TO_SCREEN_RATIO_Y = mSceneHeight / ((float)Constants::VIEW_PIXEL_HEIGHT);
109  const float TILE_WIDTH = getTileWidth();
110  const float TILE_HEIGHT = getTileHeight();
111  for(const SpriteDraw& draw : *mCurrentDraws)
112  {
113  if(!draw.pairings.empty())
114  {
115  const SpriteInfo::Info& spriteInfo = mContext.getSpriteInfo().get(draw.pairings.front().spriteId);
116 
117  const sb::tibiaassets::Object& firstObject = mContext.getObjects()[draw.pairings.front().objects.front()];
118 
119  short x = draw.topLeft.x + firstObject.itemInfo.offsetX;
120  short y = draw.topLeft.y + firstObject.itemInfo.offsetY;
121 
122  short remainderX;
123  remainderX = x % Constants::TILE_PIXEL_WIDTH;
124  if(remainderX > 0)
125  x += Constants::TILE_PIXEL_WIDTH - remainderX;
126  else
127  x -= remainderX;
128 
129  char remainderY = y % Constants::TILE_PIXEL_HEIGHT;
130  if(remainderY > 0)
131  y += Constants::TILE_PIXEL_HEIGHT - remainderY;
132  else
133  y -= remainderY;
134 
135  char tileX = x / Constants::TILE_PIXEL_WIDTH + spriteInfo.tileWidth - 1;
136  char tileY = y / Constants::TILE_PIXEL_HEIGHT + spriteInfo.tileHeight - 1;
137 
138  float screenX = mSceneX + VIEW_TEX_TO_SCREEN_RATIO_X * draw.topLeft.x + TILE_WIDTH * float(spriteInfo.tileWidth - 1);
139  float screenY = mSceneY + VIEW_TEX_TO_SCREEN_RATIO_Y * draw.topLeft.y + TILE_HEIGHT * float(spriteInfo.tileHeight - 1);
140 
141  assert(tileX >= -VISIBILITY_OFFSET_LOW && tileX < MAX_VISIBLE_TILES_X - VISIBILITY_OFFSET_LOW);
142  assert(tileY >= -VISIBILITY_OFFSET_LOW && tileY < MAX_VISIBLE_TILES_Y - VISIBILITY_OFFSET_LOW);
143 
144  Tile& tile = mData->tiles[tileX + VISIBILITY_OFFSET_LOW][tileY + VISIBILITY_OFFSET_LOW];
145  for(const SpriteDraw::SpriteObjectPairing& pair : draw.pairings)
146  {
147  assert(!pair.objects.empty());
148  for(size_t obj : pair.objects)
149  {
150 // if(obj->itemInfo.isGround)
151 // {
152 // tile = Tile();
153 // tile.tileX = tileX;
154 // tile.tileY = tileY;
155 // }
156 
157  Object o;
158  o.tileX = tileX;
159  o.tileY = tileY;
160  o.screenX = screenX;
161  o.screenY = screenY;
162  o.layer = tile.numLayers;
163  o.object = obj;
164 
165  tile.knownLayerObjects.push_back(o);
166  }
167  }
168  tile.numLayers++;
169  }
170  }
171 
172  for(size_t x = 0; x < MAX_VISIBLE_TILES_X; x++)
173  for(size_t y = 0; y < MAX_VISIBLE_TILES_Y; y++)
174  {
175  Tile& tile = mData->tiles[x][y];
176  for(Object& object : tile.knownLayerObjects)
177  object.layer = (tile.numLayers - 1) - object.layer;
178  }
179 
180  mIsCurrentFrameParsed = true;
181 }

234 void Scene::updateMovement(short x, short y)
235 {
236  short movementX = mPreviousMiniMapX - x;
237  if(movementX)
238  {
239  if(movementX > Constants::MINI_MAP_PIXEL_WIDTH - MAX_MOVEMENT)
240  movementX -= Constants::MINI_MAP_PIXEL_WIDTH;
241  else if(movementX < MAX_MOVEMENT - Constants::MINI_MAP_PIXEL_WIDTH)
242  movementX += Constants::MINI_MAP_PIXEL_WIDTH;
243  }
244 
245  short movementY = mPreviousMiniMapY - y;
246  if(movementY)
247  {
248  if(movementY > Constants::MINI_MAP_PIXEL_HEIGHT - MAX_MOVEMENT)
249  movementY -= Constants::MINI_MAP_PIXEL_HEIGHT;
250  else if(movementY < MAX_MOVEMENT - Constants::MINI_MAP_PIXEL_HEIGHT)
251  movementY += Constants::MINI_MAP_PIXEL_HEIGHT;
252  }
253 
254  mMovementX += movementX;
255  mMovementY += movementY;
256 
257  mPreviousMiniMapX = x;
258  mPreviousMiniMapY = y;
259 }
