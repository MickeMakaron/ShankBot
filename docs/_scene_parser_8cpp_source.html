51 void SceneParser::parse(const Frame& frame)
52 {
53  mData.reset(new Data());
54  mCurrentDraws = frame.spriteDraws;
55  if(mCurrentDraws == nullptr || mCurrentDraws->empty())
56  {
57  return;
58  }
59 
60  mData->screenBounds.x = round(frame.viewX);
61  mData->screenBounds.y = round(frame.viewY);
62  mData->screenBounds.width = round(frame.viewWidth);
63  mData->screenBounds.height = round(frame.viewHeight);
64 
65  const float VIEW_TEX_TO_SCREEN_RATIO_X = mData->screenBounds.width / ((float)Constants::VIEW_PIXEL_WIDTH);
66  const float VIEW_TEX_TO_SCREEN_RATIO_Y = mData->screenBounds.height / ((float)Constants::VIEW_PIXEL_HEIGHT);
67  const float TILE_WIDTH = getTileWidth();
68  const float TILE_HEIGHT = getTileHeight();
69  const int TILE_WIDTH_I = round(TILE_WIDTH);
70  const int TILE_HEIGHT_I = round(TILE_HEIGHT);
71 
72  mData->tileWidth = TILE_WIDTH;
73  mData->tileHeight = TILE_HEIGHT;
74 
75  const SpriteDraw* groundTile = findFirstGroundTile();
76  SB_EXPECT(groundTile, !=, nullptr);
77  mData->offsetX = round(groundTile->topLeft.x + 128.f) % Constants::TILE_PIXEL_WIDTH;
78  mData->offsetY = round(groundTile->topLeft.y + + 128.f) % Constants::TILE_PIXEL_HEIGHT;
79 
80  char prevTileX = -120;
81  char prevTileY = -120;
82  bool isPrevGround = false;
83  mData->layers.resize(1);
84  for(size_t i = 0; i < mCurrentDraws->size(); i++)
85  {
86  const SpriteDraw& draw = (*mCurrentDraws)[i];
87  using AObject = sb::tibiaassets::Object;
88  std::vector<const AObject*> disputes;
89  for(const auto& pairing : draw.pairings)
90  {
91  for(const size_t objectId : pairing.objects)
92  {
93  disputes.push_back(&mContext.getObjects()[objectId]);
94  }
95  }
96  SB_EXPECT_FALSE(disputes.empty());
97  const AObject& firstObject = *disputes.front();
98  const SpriteInfo::Info& spriteInfo = mContext.getSpriteInfo().get(draw.pairings.front().spriteId);
99 
100  if(firstObject.type == AObject::Type::OUTFIT || firstObject.type == AObject::Type::PROJECTILE)
101  {
102  NonGridObject o;
103  o.drawOrder = i;
104  o.disputes = disputes;
105  float x = (draw.topLeft.x + Constants::TILE_PIXEL_WIDTH * (spriteInfo.tileWidth - 1)) * VIEW_TEX_TO_SCREEN_RATIO_X;
106  float y = (draw.topLeft.y + Constants::TILE_PIXEL_HEIGHT * (spriteInfo.tileHeight - 1)) * VIEW_TEX_TO_SCREEN_RATIO_Y;
107  o.screenBounds.x = round(x) + mData->screenBounds.x;
108  o.screenBounds.y = round(y) + mData->screenBounds.y;
109  o.screenBounds.width = TILE_WIDTH_I;
110  o.screenBounds.height = TILE_HEIGHT_I;
111  o.localBounds.x = round(draw.topLeft.x);
112  o.localBounds.y = round(draw.topLeft.y);
113  o.localBounds.width = Constants::TILE_PIXEL_WIDTH;
114  o.localBounds.height = Constants::TILE_PIXEL_HEIGHT;
115 
116  Layer& layer = mData->layers.back();
117  if(firstObject.type == AObject::Type::OUTFIT)
118  {
119  Draw d;
120  d.spriteId = draw.pairings.front().spriteId;
121  d.x = VISIBILITY_OFFSET_LOW * 2 * 32 + round(draw.topLeft.x);
122  d.y = VISIBILITY_OFFSET_LOW * 2 * 32 + round(draw.topLeft.y);
123  d.layer = mData->layers.size() - 1;
124  mData->draws.push_back(d);
125  layer.characters.push_back(o);
126  }
127  else
128  {
129  layer.projectiles.push_back(o);
130  }
131  }
132  else
133  {
134  short x = round(draw.topLeft.x);
135  short y = round(draw.topLeft.y);
136  x += short(firstObject.itemInfo.offsetX) - mData->offsetX;
137  y += short(firstObject.itemInfo.offsetY) - mData->offsetY;
138 
139  short remainderX = x % Constants::TILE_PIXEL_WIDTH;
140  short remainderY = y % Constants::TILE_PIXEL_HEIGHT;
141 
142  x += (remainderX < 0 ? -remainderX : Constants::TILE_PIXEL_WIDTH - remainderX);
143  y += (remainderY < 0 ? -remainderY : Constants::TILE_PIXEL_HEIGHT - remainderY);
144 
145  char tileX = x / Constants::TILE_PIXEL_WIDTH + spriteInfo.tileWidth - 1;
146  char tileY = y / Constants::TILE_PIXEL_HEIGHT + spriteInfo.tileHeight - 1;
147  assert(tileX >= -VISIBILITY_OFFSET_LOW && tileX < MAX_VISIBLE_TILES_X - VISIBILITY_OFFSET_LOW);
148  assert(tileY >= -VISIBILITY_OFFSET_LOW && tileY < MAX_VISIBLE_TILES_Y - VISIBILITY_OFFSET_LOW);
149 
150  bool isGround = firstObject.itemInfo.isGround || firstObject.itemInfo.topOrder == 1;
151  if(isGround)
152  {
153  if(!isPrevGround && tileX + tileY < prevTileX + prevTileY)
154  {
155  mData->layers.emplace_back();
156  }
157  }
158  Layer& layer = mData->layers.back();
159 
160  Tile& tile = layer.gridObjects[tileX + VISIBILITY_OFFSET_LOW][tileY + VISIBILITY_OFFSET_LOW];
161  Object o;
162  o.drawOrder = i;
163  if(firstObject.itemInfo.isGround && firstObject.itemInfo.topOrder == 0)
164  {
165  tile.isObscured = false;
166  tile.speed = firstObject.itemInfo.walkSpeed;
167  }
168 
169  if(firstObject.itemInfo.isBlocking && !firstObject.itemInfo.isMovable)
170  {
171  tile.isWalkable = false;
172  }
173 
174  isPrevGround = isGround;
175  Draw d;
176  d.spriteId = draw.pairings.front().spriteId;
177  d.x = VISIBILITY_OFFSET_LOW * 2 * 32 + round(draw.topLeft.x);
178  d.y = VISIBILITY_OFFSET_LOW * 2 * 32 + round(draw.topLeft.y);
179  d.layer = mData->layers.size() - 1;
180  mData->draws.push_back(d);
181  prevTileX = tileX;
182  prevTileY = tileY;
183  o.disputes = disputes;
184  tile.objects.push_back(o);
185  }
186 
187  }
188  mData->playerLayer = -1;
189  auto playerLayerIt = std::find_if(mData->layers.rbegin(), mData->layers.rend(), [](const Layer& l)
190  {
191  for(auto it = l.characters.rbegin(); it != l.characters.rend(); it++)
192  {
193  int dMiddleX = it->localBounds.x + it->localBounds.width - Constants::MIDDLE_TILE_X * Constants::TILE_PIXEL_WIDTH;
194  int dMiddleY = it->localBounds.y + it->localBounds.height - Constants::MIDDLE_TILE_Y * Constants::TILE_PIXEL_HEIGHT;
195  if(dMiddleX == dMiddleY && dMiddleX <= -8 && dMiddleX >= -32)
196  {
197  return true;
198  }
199  }
200  return false;
201  });
202 
203  mData->playerLayer = std::distance(playerLayerIt, mData->layers.rend()) - 1;
204 }

219 const SpriteDraw* SceneParser::findFirstGroundTile() const
220 {
221  assert(mCurrentDraws != nullptr);
222  for(const auto& draw : *mCurrentDraws)
223  {
224  for(const auto& pairing : draw.pairings)
225  {
226  for(const size_t& id : pairing.objects)
227  {
228  const sb::tibiaassets::Object& obj = mContext.getObjects()[id];
229  if(obj.type == sb::tibiaassets::Object::Type::ITEM && obj.itemInfo.isGround)
230  {
231  return &draw;
232  }
233  }
234  }
235  }
