104 void SequenceTree::loadFromBinaryFile(NodePtr& node, std::istream& file)
105 {
106  unsigned int numElements;
107  file.read((char*)&numElements, sizeof(numElements));
108  if(numElements > 0)
109  {
110  unsigned int* elements = new unsigned int[numElements];
111  int* numValuesIgnored = new int[numElements];
112  file.read((char*)elements, sizeof(unsigned int) * numElements);

114  for(size_t i = 0; i < numElements; i++)

189 void SequenceTree::insert(NodePtr& node, Element element)
190 {
191  if(node->level >= element.sequence->size())
192  {
193  node->elements.push_back(element);
194  return;
195  }
196 
197 
198  if(!node->elements.empty())
199  {
200  node->elements.push_back(element);
201 
202  for(auto it = node->elements.begin(); it != node->elements.end();)
203  {
204  if(node->level < it->sequence->size())
205  {
206  if(node->level > 16000)

208 
209  auto child = node->children.insert(std::make_pair(it->sequence->operator[](node->level), NodePtr(new Node(node->level + 1))));
210  insert(child.first->second, *it);

212  node->elements.erase(it++);
213  }
214  else
215  it++;
216  }
217  }
