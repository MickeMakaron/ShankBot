83 std::vector<GuiParser::SideBarWindow>::const_iterator SideBarWindowAssembler::pairWindowWithTitle(const std::vector<GuiParser::SideBarWindow>& windows, const Text& title) const
84 {
85  unsigned short x = round(title.x);
86  unsigned short y = round(title.y);
87 
88  std::vector<std::vector<GuiParser::SideBarWindow>::const_iterator> its;
89  for(auto it = windows.begin(); it != windows.end(); it++)
90  {
91  if(it->titleBar.screen.y <= y && it->titleBar.screen.x <= x)
92  {
93  its.push_back(it);
94  }
95  }
96 
97  if(its.empty())
98  {
99  return windows.end();
100  }
101 
102  return *std::min_element(its.begin(), its.end(), [x, y](const auto& w1, const auto& w2)
103  {
104  unsigned short dx1 = x - w1->titleBar.screen.x;
105  unsigned short dy1 = y - w1->titleBar.screen.y;
106  unsigned short dx2 = x - w2->titleBar.screen.x;
107  unsigned short dy2 = y - w2->titleBar.screen.y;
108 
109  return dx1 * dx1 + dy1 * dy1 < dx2 * dx2 + dy2 * dy2;
110  });
111 }

440 void SideBarWindowAssembler::parseContainer(const GuiParser::SideBarWindow& window,
441  std::vector<GuiParser::Container>& containerSlots,
442  std::vector<std::vector<const SpriteDraw*>>& guiSprites,
443  std::vector<TextParser::Container>& containerCounts)
444 {
445  mData.containers.emplace_back();
446  ContainerWindow& c = mData.containers.back();
447 
448  assignWindow(c, window);
449 
450  const auto isIcon = [&c, this](const std::vector<const SpriteDraw*>& group)
451  {
452  if(group.size() != 1)
453  {
454  return false;
455  }
456  Vertex iconTopLeft;
457  group[0]->getScreenCoords(mHalfFrameWidth, mHalfFrameHeight, iconTopLeft.x, iconTopLeft.y);
458  int dIconLeft = round(iconTopLeft.x) - c.titleBar.x;
459  int dIconTop = round(iconTopLeft.y) - c.titleBar.y;
460  return dIconLeft > 0 &&
461  dIconLeft < 6 &&
462  dIconTop > 0 &&
463  dIconTop < 6;
464  };
465 
466  SB_EXPECT_FALSE(containerCounts.empty());
467  const std::vector<Text> counts = containerCounts.back().counts;
468  containerCounts.pop_back();
469 
470  if(c.isMinimized)
471  {
472  if(!guiSprites.empty() && isIcon(guiSprites.back()))
473  {
474  guiSprites.pop_back();
475  }
476  return;
477  }
478 
479  SB_EXPECT_FALSE(containerSlots.empty());
480  const std::vector<GuiElement> slots = containerSlots.back().slots;
481  containerSlots.pop_back();
482  c.capacity = slots.size();
483  SB_EXPECT(c.capacity, >, 0);
484 
485  if(guiSprites.empty())
486  {
487  return;
488  }
489 
490  if(isIcon(guiSprites.back()))
491  {
492  guiSprites.pop_back();
493  if(guiSprites.empty())
494  {
495  return;
496  }
497  }
498 
499 
500 
501  static const int SLOT_SPACING_X = Constants::TILE_PIXEL_WIDTH + Constants::CONTAINER_SEPARATOR_PIXEL_WIDTH;
502  static const int SLOT_SPACING_Y = Constants::TILE_PIXEL_HEIGHT + Constants::CONTAINER_SEPARATOR_PIXEL_WIDTH;
503  const std::vector<const SpriteDraw*>& sprites = guiSprites.back();
504  if(sprites[0]->order > window.order)
505  {
506  return;
507  }
508 
509  const int firstSlotX = round(slots[0].draw->topLeft.x);
510  const int firstSlotY = round(slots[0].draw->topLeft.y);
511  int maxIndex = -1;
512  if(!sprites.empty())
513  {
514  int x = (round(sprites.back()->topLeft.x) - firstSlotX) / SLOT_SPACING_X;
515  int y = (round(sprites.back()->topLeft.y) - firstSlotY) / SLOT_SPACING_Y;
516  maxIndex = std::max(x + y * Constants::CONTAINER_WIDTH, maxIndex);
517  }
518  if(!counts.empty())
519  {
520  int x = (round(counts.back().localX) - firstSlotX) / SLOT_SPACING_X;
521  int y = (round(counts.back().localY) - firstSlotY) / SLOT_SPACING_Y;
522  maxIndex = std::max(x + y * Constants::CONTAINER_WIDTH, maxIndex);
523  }
524 
525  size_t numItems = maxIndex + 1;
526  if(numItems > c.capacity)
527  {
528  return;
529  }
530  c.items.resize(numItems);
531 
532  size_t iSprite = 0;
533  for(size_t i = 0; i < slots.size() && iSprite < sprites.size(); i++, iSprite++)
534  {
535  const SpriteDraw& sprite = *sprites[iSprite];
536  Vertex topLeft = slots[i].draw->topLeft;
537  Vertex botRight = slots[i].draw->botRight;
538  topLeft.x -= 0.0001f;
539  topLeft.y -= 0.0001f;
540  botRight.x += 0.0001f;
541  botRight.y += 0.0001f;
542  if(sprite.topLeft.x > topLeft.x &&
543  sprite.topLeft.y > topLeft.y &&
544  sprite.botRight.x < botRight.x &&
545  sprite.botRight.y < botRight.y)
546  {
547  c.items[i].sprite = &sprite;
548  }
549  else
550  {
551  if(sprite.topLeft.y < topLeft.y)
552  {
553  c.items.clear();
554  return;
555  }
556  int beginSlotX = (round(topLeft.x) - firstSlotX) / SLOT_SPACING_X;
557  int beginSlotY = (round(topLeft.y) - firstSlotY) / SLOT_SPACING_Y;
558  int endSlotX = (round(sprite.topLeft.x) - firstSlotX) / SLOT_SPACING_X;
559  int endSlotY = (round(sprite.topLeft.y) - firstSlotY) / SLOT_SPACING_Y;
560 
561  int dX = endSlotX - beginSlotX;
562  int dY = endSlotY - beginSlotY;
563 
564  int numSlotIndicesAway = dX + Constants::CONTAINER_WIDTH * dY;
565  if(numSlotIndicesAway == 0)
566  {
567  c.items.clear();
568  return;
569  }
570 
571  i += numSlotIndicesAway;
572  if(i >= slots.size())
573  {
574  c.items.clear();
575  return;
576  }
577 
578  topLeft = slots[i].draw->topLeft;
579  botRight= slots[i].draw->botRight;
580  topLeft.x -= 0.0001f;
581  topLeft.y -= 0.0001f;
582  botRight.x += 0.0001f;
583  botRight.y += 0.0001f;
584  if(sprite.topLeft.x > topLeft.x &&
585  sprite.topLeft.y > topLeft.y &&
586  sprite.botRight.x < botRight.x &&
587  sprite.botRight.y < botRight.y)
588  {
589  c.items[i].sprite = &sprite;
590  }
591  else
592  {
593  c.items.clear();
594  return;
595  }
596  }
597  }
598 
599  size_t iCountText = 0;
600  for(size_t i = 0; i < c.items.size() && iCountText < counts.size(); i++)
601  {
602  const Vertex& topLeft = slots[i].draw->topLeft;
603  const Vertex& botRight = slots[i].draw->botRight;
604 
605  const Text& text = counts[iCountText];
606  if(text.localX > topLeft.x &&
607  text.localY > topLeft.y &&
608  text.localX < botRight.x &&
609  text.localY < botRight.y)
610  {
611  const std::string& count = text.string;
612  SB_EXPECT_TRUE(sb::utility::isNumeric(count));
613  c.items[i].count = sb::utility::strToInt(count);
614  iCountText++;
615  }
616  }
617 
618  SB_EXPECT(iSprite, ==, sprites.size());
619  SB_EXPECT(iCountText, ==, counts.size());
620  guiSprites.pop_back();
621 
622  if(!guiSprites.empty() && isIcon(guiSprites.back()))
623  {
624  guiSprites.pop_back();
625  }
626 }
