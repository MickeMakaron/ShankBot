94 std::vector<unsigned char> SpriteReader::decompressTibiaLzma(std::string path) const
95 {
96  std::ifstream file(path, std::ios::binary | std::ios::in);
97 
98  if(!validateTibiaLzmaHeader(file))
99  {
100  std::cout << "Failed to decompress \"" << path << "\". Invalid header." << std::endl;
101  return std::vector<unsigned char>();
102  }
103 
104  size_t decompressedSize = sb::utility::readTibiaSizeIndicator(file);
105  unsigned int compressedSize = sb::utility::readTibiaSizeIndicator(file);
106 
107  unsigned char props[5]; // LZMA property bytes
108  readStream(*props, file, 5);
109 
110  readStream(compressedSize, file);
111  file.seekg(4, std::ios::cur);
112 
113  unsigned char* data = new unsigned char[compressedSize];
114  readStream(*data, file, compressedSize);
115  file.close();
116 
117  std::vector<unsigned char> decompressedFile;
118  decompressedFile.resize(decompressedSize);
119  int result = sb::lzma::uncompress(decompressedFile.data(), &decompressedSize, data, &compressedSize, props, 5);
120  delete[] data;
121  switch(result)
122  {
123  case SZ_OK:
124 
125  break;
126  case SZ_ERROR_DATA:
127  std::cout << "Failed to decompress \"" << path << "\" with error code " << SZ_ERROR_DATA << "." << std::endl
128  << "Data error." << std::endl;
129  return std::vector<unsigned char>();
130  break;
131  case SZ_ERROR_MEM:
132  std::cout << "Failed to decompress \"" << path << "\" with error code " << SZ_ERROR_DATA << "." << std::endl
133  << "Memory allocation error." << std::endl;
134  return std::vector<unsigned char>();
135  break;
136  case SZ_ERROR_UNSUPPORTED:
137  std::cout << "Failed to decompress \"" << path << "\" with error code " << SZ_ERROR_DATA << "." << std::endl
138  << "Unsupported LZMA properties." << std::endl;
139  return std::vector<unsigned char>();
140  break;
141  case SZ_ERROR_INPUT_EOF:
142  std::cout << "Failed to decompress \"" << path << "\" with error code " << SZ_ERROR_DATA << "." << std::endl
143  << "Unexpected end of file." << std::endl;
144  return std::vector<unsigned char>();
145  break;
146 
147  default:
148  std::cout << "Unimplemented result code from LzmaUncompress: " << result << std::endl;
149  return std::vector<unsigned char>();
150  break;
151  }
152 
153  return decompressedFile;
154 }

156 SpriteReader::Sprite SpriteReader::getSprite(unsigned int id) const
157 {
158  auto sheetIt = std::find_if(mCatalog.getSpriteSheets().begin(),
159  mCatalog.getSpriteSheets().end(),
160  [id](const CatalogContent::SpriteSheet& sheet)

162  return id >= sheet.firstSpriteId && id <= sheet.lastSpriteId;
163  });
164 
165  if(sheetIt == mCatalog.getSpriteSheets().end())
166  {
167  Sprite s;
168  s.pixels = nullptr;
169  s.id = 0;
170  return s;
171  }
172 
173  Sprite s;
174  s.area = sheetIt->area;
175  s.id = id;
176  CatalogContent::getSpriteTileSize(sheetIt->spriteSize, s.tileWidth, s.tileHeight);
177 
178  std::vector<unsigned char> data = decompressTibiaLzma(sheetIt->path + ".lzma");
179  QImage decompressedSpriteSheet = QImage::fromData(data.data(), data.size());
180  decompressedSpriteSheet = decompressedSpriteSheet.convertToFormat(QImage::Format_RGBA8888);
181 
182  const size_t SPRITE_SHEET_PIXEL_WIDTH = decompressedSpriteSheet.width();
183  const size_t SPRITE_SHEET_PIXEL_HEIGHT = decompressedSpriteSheet.height();
184  const size_t SPRITE_PIXEL_WIDTH = s.tileWidth * constants::TILE_PIXEL_WIDTH;
185  const size_t SPRITE_PIXEL_HEIGHT = s.tileHeight * constants::TILE_PIXEL_HEIGHT;
186  size_t spriteIndex = id - sheetIt->firstSpriteId;
187  size_t spritesPerRow = SPRITE_SHEET_PIXEL_WIDTH / SPRITE_PIXEL_WIDTH;
188  size_t x = (spriteIndex % spritesPerRow) * SPRITE_PIXEL_WIDTH;
189  size_t y = (spriteIndex / spritesPerRow) * SPRITE_PIXEL_HEIGHT;
190 
191  QImage sprite = decompressedSpriteSheet.copy(x, y, SPRITE_PIXEL_WIDTH, SPRITE_PIXEL_HEIGHT);
192  s.pixels = new unsigned char[sprite.byteCount()];
193  memcpy((void*)s.pixels, sprite.bits(), sprite.byteCount());
194 
195  return s;
196 }
