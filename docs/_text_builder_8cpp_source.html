301

308

315  backLine.maxAllowedDistanceX = std::max(short(botRightX - topLeftX), backLine.maxAllowedDistanceX);

324 void TextBuilder::insert(unsigned char character, short topLeftX, short topLeftY, short botRightX, short botRightY)
325 {
326  if(mLines.empty())
327  {
328  appendLine(character, topLeftX, topLeftY, botRightX, botRightY);
329  return;
330  }
331 
332  Line& lastLine = mLines.back();
333  if(topLeftX > lastLine.botRightX)
334  {
335  unsigned short dX = topLeftX - lastLine.botRightX;
336  if(dX > lastLine.maxAllowedDistanceX)
337  {
338  appendLine(character, topLeftX, topLeftY, botRightX, botRightY);
339  return;
340  }
341  }
342  else if(topLeftX <= (lastLine.botRightX - lastLine.maxAllowedDistanceX))
343  {
344  appendLine(character, topLeftX, topLeftY, botRightX, botRightY);
345  return;
346  }
347  else if(botRightY < lastLine.topLeftY || topLeftY > lastLine.botRightY)
348  {
349  appendLine(character, topLeftX, topLeftY, botRightX, botRightY);
350  return;
351  }
352 
353  appendCharacter(character, topLeftX, topLeftY, botRightX, botRightY);
354 
355  return;
356 }

360 void TextBuilder::build()
361 {
362  for(const GlyphDraw& glyph : *mTextDraw.glyphDraws)
363  insert(glyph.character, glyph.topLeft.x, glyph.topLeft.y, glyph.botRight.x, glyph.botRight.y);
364 
365  std::vector<Text> lines;

368  {
369  Text t;
370 
371  mTextDraw.getScreenCoords(M_HALF_FRAME_WIDTH, M_HALF_FRAME_HEIGHT, t.x, t.y, line.topLeftX, line.topLeftY);
372  t.localX = line.topLeftX;
373  t.localY = line.topLeftY;
374 // t.topLeftX = line.topLeftX;
375 // t.topLeftY = line.topLeftY;
376  t.width = line.botRightX - line.topLeftX;
377  t.height = line.botRightY - line.topLeftY;
378  t.order = mTextDraw.order;
379  t.string.reserve(line.characters.size());
380  for(const unsigned char& c : line.characters)
381  t.string.push_back(c);
382  lines.push_back(t);
383  }
384 
385  mText = lines;
386 }
