100  } while(i < mBuilders.size() && mBuilders[i]->getTextType() == T::CHAT_TAB);

102  }
103  else

105  unsigned int drawCallId = (*mDraws)[i].drawCallId;
106  do
107  {
108  i++;
109  } while(i < mBuilders.size() &&
110  (*mDraws)[i].drawCallId == drawCallId &&

117  case T::GUI:
118  handleGuiText(i);

156 void TextParser::handleDefaultSideBarText(size_t& i)
157 {
158  enum Type : unsigned char
159  {
160  CENTRE,
161  HP_MANA,
162  PREMIUM_FEATURES,
163  SOUL_CAP,
164  STOP,
165  NUM_TYPES,
166  };
167  std::set<Type> parsedTypes;
168 
169 
170  using namespace sb::utility;
171  while(i < mBuilders.size() && parsedTypes.size() < NUM_TYPES)
172  {
173  const std::vector<Text>& text = mBuilders[i]->getText();
174  auto it = text.begin();
175  if(it->string == "soul:" || it->string == "CaP:")
176  {
177  SB_EXPECT(text.size(), ==, 2);
178  i += 2;
179  SB_EXPECT(i, <, mBuilders.size());
180  mData.isInventoryMinimized = it->string == "CaP:";
181  if(mData.isInventoryMinimized)
182  {
183  SB_EXPECT(text[1].string, ==, "soul:");
184  readSoulAndCap(*mBuilders[i], *mBuilders[i - 1]);
185  }
186  else
187  {
188  SB_EXPECT(text[1].string, ==, "CaP:");
189  readSoulAndCap(*mBuilders[i - 1], *mBuilders[i]);
190  }
191 
192  parsedTypes.insert(Type::SOUL_CAP);
193  it = text.end();
194  }
195  else if(isNumeric(it->string))
196  {
197  if(it->type == Text::Type::ITEM_STACK_COUNT)
198  {
199  mData.inventoryStackCounts.insert(mData.inventoryStackCounts.end(), text.begin(), text.end());
200  it = text.end();
201  }
202  else
203  {
204  SB_EXPECT(text.size(), >=, 2);
205 
206  SB_EXPECT_TRUE(isNumeric(it->string));
207  mData.hp = strToInt(it->string);
208 
209  SB_EXPECT(++it, !=, text.end());
210  SB_EXPECT_TRUE(isNumeric(it->string));
211  mData.mana = strToInt(it->string);
212 
213  it++;
214  parsedTypes.insert(HP_MANA);
215  }
216  }
217 
218  while(it != text.end())
219  {
220  if(it->string == "Centre")
221  {
222  mData.centre = *it;
223  parsedTypes.insert(CENTRE);
224  }
225  else if(it->string == "Stop")
226  {
227  mData.stop = *it;
228  parsedTypes.insert(STOP);
229  }
230  else if(it->string == "Premium Features")
231  {
232  parsedTypes.insert(PREMIUM_FEATURES);
233  mData.isPremiumFeaturesMinimized = (std::distance(it, text.end()) == 1);
234  if(!mData.isPremiumFeaturesMinimized)
235  {
236  break;
237  }
238  }
239  else
240  {
241  SB_THROW("Unexpected text. Got \"" + it->string + "\".");
242  }
243  it++;
244  }
245  i++;
246  }
247  i--;
248 
249  SB_EXPECT(parsedTypes.size(), ==, NUM_TYPES);
250 }
