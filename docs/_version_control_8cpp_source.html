117 bool VersionControl::updateVersion(std::string clientDir, std::string versionControlDir)
118 {
119  std::string currentVersionPath = versionControlDir + CURRENT_VERSION_PATH;
120  if(!fileExists(currentVersionPath))
121  {
122  mVersionFile.version = getTibiaVersion(clientDir);
123  return true;
124  }
125 
126  VersionFile versionFile = VersionFile::fromFile(currentVersionPath);
127  std::string tibiaVersion = getTibiaVersion(clientDir);
128  if(versionFile.version == tibiaVersion)
129  return false;
130 
131  mVersionFile.version = tibiaVersion;
132 
133  std::string currentVersionDir = sb::utility::file::getPath(currentVersionPath);
134  std::string oldVersionsDir = versionControlDir + OLD_PATH;
135  backup(oldVersionsDir, currentVersionDir);
136  recursiveRemove(currentVersionDir, false);
137  return true;
138 }

146 std::string VersionControl::getTibiaVersion(std::string clientDir)
147 {
148  std::string packageInfoFilePath = clientDir + TIBIA_PACKAGE_INFO_PATH;
149 
150  QFile file(QString::fromStdString(packageInfoFilePath));
151  file.open(QIODevice::ReadOnly | QIODevice::Text);
152  QString content = file.readAll();
153  file.close();
154  QJsonDocument package = QJsonDocument::fromJson(content.toUtf8());
155 
156  if(package.isNull())
157  SB_THROW("Failed to read Tibia package at '" + packageInfoFilePath + "'.");
158 
159  QJsonObject obj = package.object();
160  return obj["version"].toString().toStdString();
161 }
