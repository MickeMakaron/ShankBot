59 RequestResult Requester::frame(Frame& f)
60 {
61  RequestResult result;
62  auto response = mRequester->request(result, std::make_unique<msg::FrameRequest>());
63  if(response == nullptr)
64  return result;
65  assert(msg::Response::readResponseType(response->data(), response->size()) == msg::Message::Type::FRAME_RESPONSE);
66  msg::FrameResponse r;
67  if(!r.fromBinary(response->data(), response->size()))
68  SB_THROW("Failed to read frame response.");
