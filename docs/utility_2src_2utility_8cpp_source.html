379 size_t movingWindowMinDiff
380 (
381  const unsigned char* lhs,
382  size_t lhsWidth, size_t lhsHeight,
383  const unsigned char* rhs,
384  size_t rhsWidth, size_t rhsHeight,
385  size_t rhsSum
386 )
387 {
388  if(lhsWidth > rhsWidth || lhsHeight > rhsHeight)
389  SB_THROW("LHS size cannot be higher than RHS size.");
390 
391  const size_t NUM_ITERATIONS_X = rhsWidth - lhsWidth + 1;
392  const size_t NUM_ITERATIONS_Y = rhsHeight - lhsHeight + 1;
393  const size_t LHS_ROW_SIZE = lhsWidth;
394  const size_t RHS_ROW_SIZE = rhsWidth;
395  const size_t LHS_SIZE = lhsHeight * LHS_ROW_SIZE;
396  const size_t RHS_SIZE = rhsHeight * RHS_ROW_SIZE;
397  size_t minDiff = -1;
398 
399  if(rhsSum == -1)
400  {
401  rhsSum = 0;
402  for(size_t i = 0; i < RHS_SIZE; i++)
403  {
404  rhsSum += rhs[i];
405  }
406  }
407 
408  for(size_t x = 0; x < NUM_ITERATIONS_X; x++)
409  for(size_t y = 0; y < NUM_ITERATIONS_Y; y++)
410  {
411  size_t diff = 0;
412  size_t currentRhsSum = rhsSum;
413  for(size_t lhsIndex = 0, rhsIndex = x + y * RHS_ROW_SIZE;
414  lhsIndex < LHS_SIZE && rhsIndex < RHS_SIZE;
415  lhsIndex += LHS_ROW_SIZE, rhsIndex += RHS_ROW_SIZE)
416  {
417  for(size_t i = 0; i < LHS_ROW_SIZE; i++)
418  {
419  currentRhsSum -= rhs[rhsIndex + i];
420  int d = (int)lhs[lhsIndex + i] - (int)rhs[rhsIndex + i];
421  diff += d < 0 ? -d : d;
422  }
423  }
424 
425  diff += currentRhsSum;
426  if(diff == 0)
427  return 0;
428 
429  if(diff < minDiff)
430  minDiff = diff;
431  }
432 
433  return minDiff;
434 }
